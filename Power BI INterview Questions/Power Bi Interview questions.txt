==> How do you find out the Duplicates and Deletes in SQL?
A: Find column wise:

   Select Column1, Column2, Count(*) As Duplicates
   from Table t1
   Group by Column1, column2
   Having Count(*) > 1   
   
   Find All the table;

   Select * From table t1
   Where Exists(Select 1 from Table t2
        Where t1.column1 = t2.column1
        And t1.column2 = t2.column2
		And t1.id > t2.id    
		
   Delete Duplicates:
   
   With CTE AS(
      Select *,ROW_NUMBER, OVER(Partition by column1,column2,order by id),rid from table
) 
Delete from CTE where rid > 1
